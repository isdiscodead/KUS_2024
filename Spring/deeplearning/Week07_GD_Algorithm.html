x = 10
learning_rate = 0.2 # eta
precision = 0.00001
max_iterations = 100

# loss function is defined with lambda function.
loss_func = lambda x: (x-3)**2 + 10

# gradient is defined with lambda function. It is the 1st derivative of loss function
gradient = lambda x: 2*x-6

# gradient descent
for i in range(max_iterations):
    x = x - learning_rate * gradient(x)
    print("loss(", x, ")=", loss_func(x))
    
print("minimum = ", x)

Multivariate Gradient - 2-dimensional gradient¶

gradient value and update rule

∇𝑓(𝑥) = [𝜕𝑓(𝑥)/𝜕𝑥1, 𝜕𝑓(𝑥)/𝜕𝑥2, … , 𝜕𝑓(𝑥)/𝜕𝑥𝑑]^𝑇

𝒙^+ = 𝒙 − 𝜂∇𝑓(𝒙) <- x^+는 vector이므로 ∇𝑓(𝒙)도 vector여야 함

Loss Function 𝑧 = 𝑓(𝑥, 𝑦) = 𝑥^2 + 𝑦^2

import matplotlib.pyplot as plt
import numpy as np

x = np.arange(-5,5,0.5) # 더 큰 범위 사용 시 실수 범위처럼 넓어짐
y = np.arange(-5,5,0.5)

X, Y = np.meshgrid(x,y)
U = -2*X # gradient w.r.t. X
V = -2*Y # gradient w.r.t. Y

plt.figure()

Q = plt.quiver(X, Y, U, V, units='width') # take the values (U, V) and show as vectors

plt.show() # arrow = gradient ( value, direction )
